public class SyncExample {
    private int counter = 0;

    public void thread1() {
        synchronized (this) {
            counter++;
        }
    }

    public void thread2() {
        synchronized (this) {
            counter++;
        }
    }

    public int getCounter() {
        return counter;
    }
}
