import java.util.concurrent.locks.ReentrantLock;

public class IntegerExample {
    private int counter = 0;
    private final ReentrantLock lock = new ReentrantLock();

    public void thread1() {
        lock.lock();
        try {
            counter++;
        } finally {
            lock.unlock();
        }
    }

    public void thread2() {
        lock.lock();
        try {
            counter++;
        } finally {
            lock.unlock();
        }
    }

    public int getCounter() {
        return counter;
    }
}
