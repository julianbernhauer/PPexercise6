import java.util.concurrent.atomic.AtomicInteger;

public class IntegerExample {
    private AtomicInteger counter = new AtomicInteger(0);

    public void thread1() {
        counter.incrementAndGet();
    }

    public void thread2() {
        counter.incrementAndGet();
    }

    public int getCounter() {
        return counter.get();
    }
}
